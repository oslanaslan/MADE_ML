def train(dataloader, model, loss_func, optimizer):
    '''Train model'''
    size = len(dataloader.dataset)
    train_loss, correct = 0, 0
    model.train()
    
    for batch_idx, (X_data, y_data) in enumerate(dataloader):
        X_data = X_data.to(device)
        y_data = y_data.to(device)
        
        # Forward
        pred = model(X_data)
        loss = loss_func(pred, y_data)
        
        # Backward
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        
        train_loss += loss_func(pred, y_data).item()
        correct += (pred.argmax(1) == y_data).type(torch.float).sum().item()
        
    train_loss /= size
    correct /= size
    
    return correct, train_loss
            
def test(dataloader, model, loss_func):
    '''Test accuracy'''
    size = len(dataloader.dataset)
    model.eval()
    test_loss, correct = 0, 0
    with torch.no_grad():
        for X_data, y_data in dataloader:
            X_data = X_data.to(device)
            y_data = y_data.to(device)
            pred = model(X_data)
            test_loss += loss_func(pred, y_data).item()
            correct += (pred.argmax(1) == y_data).type(torch.float).sum().item()
    test_loss /= size
    correct /= size
    
    return correct, test_loss

def plot_data(data: list, title: str, legend: list, xlabel: str, ylabel: str) -> None:
    '''Plot data from list'''
    plt.title(title)
    plt.grid()

    for data_item in data:
        x_range_lst = list(range(len(data_item)))
        plt.plot(x_range_lst, data_item)
        
    plt.legend(legend)
    plt.xlabel(xlabel)
    plt.ylabel(ylabel)
    plt.show()

def train(dataloader, model, loss_func, optimizer):
    '''Train model'''
    max_batch_count = len(dataloader) / 4
    batch_size = 128
    # size = int(batch_size * max_batch_count)
    size = 0
    train_loss, correct = 0, 0
    model.train()
    
    for batch_idx, (X_data, y_data) in enumerate(dataloader):
        if batch_idx >= max_batch_count:
            break
        X_data = X_data.to(device)
        y_data = y_data.to(device)
        
        # Forward
        pred = model(X_data)
        loss = loss_func(pred, y_data)
        
        # Backward
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        
        train_loss += loss_func(pred, y_data).item()
        correct += (pred.argmax(1) == y_data).type(torch.float).sum().item()
        size += pred.shape[0]
        
    train_loss /= size
    correct /= size
    
    return correct, train_loss
